import { useState } from "react";

function ResumeUpload() {
  const [resume, setResume] = useState(null); //eslint-disable-line
  const [_, setDragging] = useState(false); //eslint-disable-line

  // const handleFileChange = (event) => {

  const handleFileChange = (event) => {
    const file = event.target.files[0];
    setResume(file);
    handleSubmit(file);
  };

  const handleDragOver = (event) => {
    event.preventDefault();
    setDragging(true);
  };

  const handleDragLeave = () => {
    setDragging(false);
  };

  // const handleDrop = (event) => {

  //     handleSubmit(file); // Automatically submit upon file drop;
  //   }
  // };

  const handleDrop = (event) => {
    event.preventDefault();
    setDragging(false);
    if (event.dataTransfer.items && event.dataTransfer.items[0]) {
      const file = event.dataTransfer.items[0].getAsFile();
      setResume(file);
      handleSubmit(file);
    }
  };

  // const handleSubmit = () => {

  //   } else {
  //     console.log("No file selected");
  //   }
  // };

  const handleSubmit = (file) => {
    if (file) {
      console.log("File selected: ", file.name);
      // Later here we will make an API call to the backend
    } else {
      console.log("No file selected");
    }
  };

  const openFileDialog = () => {
    document.getElementById("fileInput").click();
  };

  return (
    <div
      onClick={openFileDialog}
      onDragOver={handleDragOver}
      onDragLeave={handleDragLeave}
      onDrop={handleDrop}
      style={{
        border: "2px dashed #000",
        padding: "20px",
        width: "300px",
        textAlign: "center",
        cursor: "pointer",
      }}
    >
      <input
        type="file"
        id="fileInput"
        onChange={handleFileChange}
        style={{ display: "none" }}
      />
      {/* <button onClick={handleUploadClick}>Upload Resume</button> */}

    </div>
  );
}
